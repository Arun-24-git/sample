#!/bin/bash

if [ $# -ne 2 ]; then
  echo "Syntax: $0 <file> <word to delete>"
  exit 1
fi

file="$1"
word="$2"

if grep -qw "$word" "$file"; then
  # Create a temporary file without lines containing the word
  grep -wv "$word" "$file" > "${file}.tmp" && mv "${file}.tmp" "$file"
  echo "Word '$word' was removed from '$file'."
else
  echo "Error: the word '$word' was not found in '$file'."
fi

exit 0

#!/bin/sh

if [ $# -gt 1 ]; then
  echo "Syntax is <$0> [<file>]"
  exit 1
fi

nol=0
nowds=0
flag=0

if [ $# -eq 1 ]; then
  term=$(tty)
  exec < "$1"
  flag=1
fi

while read line; do
  nol=$((nol + 1))
  set -- $line
  nowds=$((nowds + $#))
done

echo "number of lines = $nol"
echo "number of words = $nowds"

if [ $flag -eq 1 ]; then
  exec < "$term"
fi

exit 0
